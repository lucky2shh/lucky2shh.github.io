1、环境准备

1.1、建表

CREATE TABLE s2_student(
  sno VARCHAR(3) NOT NULL,
  sname VARCHAR(4) NOT NULL,
  sex VARCHAR(2) NOT NULL,
  birthday DATETIME,
  class VARCHAR(5)
);

CREATE TABLE s2_teacher(
  tno VARCHAR(3) NOT NULL,
  tname VARCHAR(4) NOT NULL,
  sex VARCHAR(2)  NOT NULL,
  birthday DATETIME NOT NULL,
  prof VARCHAR(6),
  depart VARCHAR(10) NOT NULL
);

CREATE TABLE s2_course(
  cno VARCHAR(5) NOT NULL,
  cname VARCHAR(10) NOT NULL,
  tno VARCHAR(10) NOT NULL
);

CREATE TABLE s2_score(
  sno VARCHAR(3) NOT NULL,
  cno VARCHAR(5) NOT NULL,
  degree NUMERIC(10, 1) NOT NULL
);

2、练习

2.1、查询Score表中成绩在60到80之间的所有记录

select s.*
from s2_score as s
where s.degree BETWEEN 60 and 80;

2.2、查询 score 表中成绩为85，86或88的记录

select s.*
from s2_score as s
where s.degree = 85 or s.degree = 86 or s.degree = 88;

2.3、以 cno 升序、degree降序查询 score 表的所有记录

select s.*
from s2_score as s
order by s.cno asc, s.degree desc;

2.4、查询“95031”班的学生人数。

select COUNT(*) as "人数"
from s2_student as st
where st.class = "95031";

2.5、查询Score表中的最高分的学生学号和课程号

select
	s.sno,
	s.cno,
	s.degree
from s2_score as s
order by s.degree desc
limit 0, 1;

2.6、查询‘3-105’号课程的平均分

select
	s.cno,
	avg(s.degree) as "平均分"
from s2_score as s
where s.cno = "3-105"
group by s.cno

2.7、查询Score表中至少有5名学生选修的并以3开头的课程的平均分数

select
	s.cno,
	avg(s.degree) as "平均分"
from s2_score as s
where s.cno like "3%"
group by s.cno
having count(s.sno) >= 5

2.8、查询最低分大于70，最高分小于90的Sno列

select
	s.sno
from s2_score as s
where s.degree > 70 and s.degree < 90

2.9、查询所有学生的Sname、Cno和Degree列

select
	st.sname,
	o1.cno,
	o1.degree
from s2_student as st
join (
	select s.*
	from s2_score as s
) as o1 on st.sno = o1.sno

2.10、查询所有学生的Sno、Cname和Degree列

select
	o1.sno,
	c.cname,
	o1.degree
from s2_course as c
join (
	select s.*
	from s2_score as s
) as o1 on c.cno = o1.cno

2.11、查询所有学生的Sname、Cname和Degree列

select
	st.sname,
	c.cname,
	o1.degree
from s2_course as c
join (
	select s.*
	from s2_score as s
) as o1 on c.cno = o1.cno
join s2_student as st on o1.sno = st.sno

2.12、查询“95033”班所选课程的平均分

select
	o2.cno,
	avg(o2.degree) as "平均分"
from (
	select s.*
	from s2_score as s
	join (
		select st.sno
		from s2_student as st
		where st.class = "95033"
	) as o1 on s.sno = o1.sno
) as o2
group by o2.cno

2.13、查询所有同学的Sno、Cno和评级。（60分以下级别为E，60-70为D，70-80为C，80-90为B，90-100为A）

select
	s.sno,
	s.cno,
	s.degree,
	IF(s.degree <= 100 and s.degree > 90, "A",
	IF(s.degree <= 90 and s.degree > 80, "B", 
	IF(s.degree <= 80 and s.degree > 70, "C",
	IF(s.degree <= 70 and s.degree > 60, "D",
	IF(s.degree <= 60, "E", ""))))) as "评级"
from s2_score as s
order by s.degree desc

2.14、查询选修“3-105”课程的成绩高于“109”号同学成绩的所有同学的记录

select s.*
from s2_score as s
where s.cno = "3-105" and s.degree > (
	select degree from s2_score where cno = "3-105" and sno = "109"
)

2.15、查询score中选学一门以上课程的同学中分数为非最高分成绩的记录

select
	o2.sno,
	o2.cno,
	o2.degree
from (
	select
		s.*,
		ROW_NUMBER() over (order by s.degree desc) as "d_level"
	from s2_score as s
	join (
		select
			s.sno
		from s2_score as s
		group by s.sno
		having count(s.cno) > 1
	) as o1 on s.sno = o1.sno
) as o2
where o2.d_level > 1

2.16、查询和学号为108的同学同年出生的所有学生的Sno、Sname和Sbirthday列

select
	st.sno,
	st.sname,
	st.birthday
from s2_student as st
where YEAR(st.birthday) = (
	select YEAR(birthday) from s2_student where sno = "108"
) and st.sno != "108"

2.17、查询“张旭“教师任课的学生成绩

select s.*
from s2_score as s
join (
	select c.cno
	from s2_course as c
	join (
		select t.tno
		from s2_teacher as t
		where t.tname = "张旭"
	) as o1 on c.tno = o1.tno
) as o2 on s.cno = o2.cno

2.18、查询选修某课程的同学人数多于5人的教师姓名

select t.*
from s2_teacher as t
join (
	select c.tno
	from s2_course as c
	join (
		select s.cno
		from s2_score as s
		group by s.cno
		having count(s.sno) > 5
	) as o1 on c.cno = o1.cno
) as o2 on t.tno = o2.tno

2.19、查询95033班和95031班全体学生的记录

select st.*
from s2_student as st
where st.class = "95033" or st.class = "95031"

2.20、查询存在有85分以上成绩的课程Cno

select DISTINCT s.cno
from s2_score as s
where s.degree > 85

2.21、查询出“计算机系“教师所教课程的成绩表

select s.*
from s2_score as s
join (
	select c.cno
	from s2_course as c
	join (
		select t.tno
		from s2_teacher as t
		where t.depart = "计算机系"
	) as o1 on c.tno = o1.tno
) as o2 on s.cno = o2.cno

2.22、查询所有教师和同学的name、sex和birthday

select
	st.sname,
	st.sex,
	st.birthday
from s2_student as st
union all
select
	t.tname,
	t.sex,
	t.birthday
from s2_teacher as t

2.23、查询所有“女”教师和“女”同学的name、sex和birthday

select
	st.sname,
	st.sex,
	st.birthday
from s2_student as st
where st.sex = "女"
union all
select
	t.tname,
	t.sex,
	t.birthday
from s2_teacher as t
where t.sex = "女"

2.24、查询成绩比该课程平均成绩低的同学的成绩表

select o2.*
from (
	select
		s.*,
		o1.degree_avg
	from s2_score as s
	join (
		select
			s.cno,
			avg(s.degree) as "degree_avg"
		from s2_score as s
		group by s.cno
	) as o1 on s.cno = o1.cno
) as o2
where o2.degree < o2.degree_avg

2.25、查询所有任课教师的Tname和Depart

select
	t.tname,
	t.depart
from s2_teacher as t
join (
	select c.tno
	from s2_course as c
) as o1 on t.tno = o1.tno

2.26、查询至少有2名男生的班号

select
	st.class
from s2_student as st
where st.sex = "男"
group by st.class
having count(st.sex) >= 2

2.27、查询Student表中不姓“王”的同学记录

select st.*
from s2_student as st
where st.sname not like "王%";

2.28、查询Student表中每个学生的姓名和年龄

select
	st.sno,
	st.sname,
	(YEAR(NOW()) - YEAR(st.birthday)) as "年龄"
from s2_student as st
