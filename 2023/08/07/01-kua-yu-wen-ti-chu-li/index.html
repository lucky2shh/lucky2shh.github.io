微服务问题系列-处理跨域

微服务问题 解决跨域
微服务
发布日期:   2023-08-07
更新日期:   2023-08-07
文章字数:   442

1、跨域问题

只要 访问协议、访问地址、端口号 三个地方有一个不一样就会出现跨域问题。

2、跨域解决方案

2.1、注解

可以在Controller控制器上加上注解@CrossOrigin即可，不过这种方案生产中不推荐，仅在开发测试阶段来使用。

2.2、网关

可以通过网关SpringGateway来解决跨域，生产中建议使用该解决方案。

2.2.1、配置文件方案

在配置文件中加入如下内容即可：

spring:
  cloud:
    gateway:
      globalcors: # 全局的跨域处理
        add-to-simple-url-handler-mapping: true # 设置为true来解决options请求被拦截问题
        corsConfigurations:
          '[/**]':
            allowedOrigins: # 允许哪些网站的跨域请求 
              - "http://localhost:8090"
            allowedMethods: # 允许的跨域ajax的请求方式
              - "GET"
              - "POST"
              - "DELETE"
              - "PUT"
              - "OPTIONS"
            allowedHeaders: "*" # 允许在请求中携带的头信息
            allowCredentials: true # 是否允许携带cookie
            maxAge: 360000 # 跨域检测有效期

路由转发配置内容示例如下：

spring:
  cloud:
    nacos: # nacos配置，可以选择nacos作为配置中心，维护更方便。
      discovery:
        server-addr: 127.0.0.1:8848
        namespace: f2466630-4cfd-4f5c-975d-47b6a3d81a0a
        group: syt-service
    gateway:
      routes:
        - id: syt-user-router
          uri: lb://service-user
          predicates:
            - Path=/api/user/**
          filters:
            - RewritePath=/api/?(?<segment>.*), /$\{segment}
        - id: syt-hosp-router
          uri: lb://service-hosp
          predicates:
            - Path=/api/hosp/**
          filters:
            - RewritePath=/api/?(?<segment>.*), /$\{segment}
        - id: syt-cmn-router
          uri: lb://service-cmn
          predicates:
            - Path=/api/cmn/**
          filters:
            - RewritePath=/api/?(?<segment>.*), /$\{segment}

2.2.2、配置类方案

这里注意，配置类方案与配置文件方案 二选一即可。

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.cors.CorsConfiguration;
import org.springframework.web.cors.reactive.CorsWebFilter;
import org.springframework.web.cors.reactive.UrlBasedCorsConfigurationSource;

@Configuration
public class SYTCorsConfig {
    @Bean
    public CorsWebFilter corsWebFilter(){
        CorsConfiguration corsConfiguration = new CorsConfiguration();
        corsConfiguration.addAllowedHeader("*");
        corsConfiguration.addAllowedMethod("*");
        corsConfiguration.addAllowedOrigin("*");
        //corsConfiguration.setAllowCredentials(true);
        UrlBasedCorsConfigurationSource configurationSource = new UrlBasedCorsConfigurationSource();
        configurationSource.registerCorsConfiguration("/**", corsConfiguration);
        return new CorsWebFilter(configurationSource);
    }
}
